408 /***********************************************************************
409 * Type: errnum_t
410 * Error code. 0=NoError.
411 *
412 * Description:
413 * Error code is defined in *_errnum.h file.
414 * Bit field in the error node is:
415 *
416 * > 3 2 1
417 * > 1 4 6 8 0
418 * > ---------------------------------------
419 * > 0GGG GGGG MMMM MMMM SSSS SSSS IIII IIII
420 * >
421 * > 0 - 0 only
422 * > G - Defined in global error code
423 * > M - Defined in each module. ID of the module
424 * > S - Defined in each module. Error code of the module
425 * > I - Defined in each module. Internal ID
426 *
427 * Error code must be compared with mask.
428 * Because internal ID must be ignored.
429 * > if ( ( e & E_FEW_MEMORY_MASK ) == E_FEW_MEMORY ) { }
430 ************************************************************************/
431 typedef int errnum_t;
